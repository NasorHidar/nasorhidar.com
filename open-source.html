# Before: missing timestamp fields
class FindingFilter(FilterSet):
  # created_before / after — absent
  # modified timestamps — absent

# After: Nasor's patch PR #14754
class FindingFilter(FilterSet):
  created = DateTimeFromToRangeFilter(
    field_name='date'
  )
  created_before = DateTimeFilter(
    field_name='date',
    lookup_expr='lte'
  )
  created_after = DateTimeFilter(
    field_name='date',
    lookup_expr='gte'
  )
